Fetch Official Results (fails gracefully if not found) let official = []; try { const offRes = await fetch(officialBracketURL); if (offRes.ok) official = await offRes.json(); } catch (e) { console.warn("Official results not found."); } // 2. Parse CSV Rows const rows = csvText.split('\n').map(row => row.trim()).filter(row => row !== ""); const users = []; // Start at index 1 to skip headers for (let i = 1; i < rows.length; i++) { // This regex handles commas inside quotes const columns = rows[i].split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/); const name = columns[1] ? columns[1].replace(/"/g, "") : "Anonymous"; let bracketData = []; try { // Clean up Google Sheets' triple-quote mess for JSON let rawJson = columns[2] || "[]"; rawJson = rawJson.replace(/^"|"$/g, '').replace(/""/g, '"'); bracketData = JSON.parse(rawJson); } catch (e) { console.error("JSON Parse error for", name); } users.push({ name, rounds: bracketData }); } // 3. Run Calculations const scores = calculateScores(users, official); // 4. Render Table const tbody = document.getElementById("leaderboard-body"); tbody.innerHTML = ""; scores.forEach((user, i) => { const tr = document.createElement("tr"); tr.innerHTML = ` ${i + 1} ${user.name} ${user.score} ${user.roundsCorrect} `; tbody.appendChild(tr); }); } catch (err) { console.error("Leaderboard error:", err); } } function calculateScores(users, official) { const scores = []; function normalize(str) { if (!str) return ""; return str.toString().toLowerCase().replace(/^\d+\s*/, "").replace(/[^a-z0-9]/g, "").trim(); } users.forEach(user => { let score = 0; let roundsCorrect = 0; // Only calculate if official results exist if (official && official.length > 0) { official.forEach((round, rIndex) => { if (!user.rounds || !user.rounds[rIndex]) return; const roundPoints = Math.pow(2, rIndex); const userMatchups = user.rounds[rIndex]; round.matchups.forEach((offMatch, mIndex) => { const officialWinner = offMatch?.name; const userPick = userMatchups?.[mIndex]?.name; // Adjust based on your JSON structure if (officialWinner && userPick && normalize(userPick) === normalize(officialWinner)) { score += roundPoints; roundsCorrect++; } }); }); } scores.push({ name: user.name, score, roundsCorrect }); }); return scores.sort((a, b) => b.score - a.score); } // Start renderBracket(); updateLeaderboard(); setInterval(updateLeaderboard, 30000);
